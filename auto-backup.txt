function backupDatabase() {
 const databasePath = './lib/database/file_yang_ingin_di_backup.json';
 
 const backupPath = `./lib/database/data_backup.json`;
 fs.readFile(databasePath, 'utf8', (err, data) => {
 if (err) {

 return;
 }

 fs.writeFile(backupPath, data, (err) => {
 if (err) {
 console.error('Gagal membuat backup:', err);
 return;
 }
 console.log('Backup database berhasil!');
 sendBackupToOwner(backupPath);
 });
 });
